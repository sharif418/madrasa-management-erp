  students      Student[]
  staff         Staff[]

  academicLevels AcademicLevel[]

  hostels       Hostel[]

model Campus {
  id        String   @id @default(cuid())
  tenantId  String
  name      String
  address   String?
  phone     String?
  isMain    Boolean  @default(false)
  createdAt DateTime @default(now())

  tenant    Tenant   @relation(fields: [tenantId], references: [id], onDelete: Cascade)
  students  Student[]
  staff     Staff[]
  hostels   Hostel[]

}

model FeatureToggle {
  id        String  @id @default(cuid())
  tenantId  String
  module    String  // RESIDENTIAL | TRANSPORT | HIFZ | FINANCE_ZAKAT | HR | MESS | MUHASABA | WEBSITE | AI
  enabled   Boolean @default(true)
  tenant    Tenant  @relation(fields: [tenantId], references: [id], onDelete: Cascade)

  @@unique([tenantId, module])
}

// ============================================================
// 2. RBAC — ROLES, PERMISSIONS, USERS
// ============================================================

model User {
  id        String   @id @default(cuid())
  tenantId  String
  email     String   @unique
  name      String
  phone     String?
  passwordHash String?
  avatarUrl String?
  role      String   @default("TEACHER") // SUPER_ADMIN | PRINCIPAL | TEACHER | WARDEN | ACCOUNTANT | PARENT | STUDENT
  customRole String? // dynamic role name
  isActive  Boolean  @default(true)
  lastLoginAt DateTime?
  createdAt DateTime @default(now())
  updatedAt DateTime @updatedAt

  tenant    Tenant   @relation(fields: [tenantId], references: [id], onDelete: Cascade)
  permissions Permission[]
  hifzLogs   HifzLog[]
  staffProfile Staff?
  parentProfile Parent?
  studentProfile Student?
  notificationsSent Notification[]
}

model Permission {
  id          String  @id @default(cuid())
  userId      String
  module      String  // STUDENTS | FINANCE | HIFZ | HOSTEL | HR | MESS | MUHASABA | SETTINGS
  action      String  // VIEW | EDIT | EXPORT | OVERRIDE
  scope       String  @default("ALL") // ALL | OWN_CLASS | OWN_DEPT
  user        User    @relation(fields: [userId], references: [id], onDelete: Cascade)

  @@unique([userId, module, action])
}
